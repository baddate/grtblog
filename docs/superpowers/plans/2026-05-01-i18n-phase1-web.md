# i18n Phase 1: Web (SvelteKit) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build i18n infrastructure for the web frontend and migrate ~600 hardcoded Chinese strings to a translation system supporting zh/en via URL-prefixed routes.

**Architecture:** Shared common locale files at repo root, web-specific locales in `web/src/locales/`. Build-time merge script produces a single flat JSON per language. SvelteKit `hooks.server.ts` detects language from URL `[lang]` param, loads translations, injects `t()` via `event.locals`. Footer language switcher updates Cookie and redirects.

**Tech Stack:** SvelteKit 2.57, Svelte 5 runes, custom lightweight i18n (no external library), Vitest for tests

**Design spec:** `docs/superpowers/specs/2026-05-01-i18n-design.md`

### Task 3: Create locale merge script with type generation

**Files:**
- Create: `web/scripts/merge-locales.mjs`

- [ ] **Step 1: Create `web/scripts/merge-locales.mjs`**

```javascript
import { readFileSync, writeFileSync, mkdirSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = join(__dirname, '..');
const REPO_ROOT = join(ROOT, '..', '..');
const OUT_DIR = join(ROOT, 'src', 'lib', 'i18n', 'generated');

const LANGS = ['zh', 'en'];

function readJSON(path) {
  return JSON.parse(readFileSync(path, 'utf-8'));
}

function mergeLocales(lang) {
  const common = readJSON(join(REPO_ROOT, 'locales', lang, 'common.json'));
  const web = readJSON(join(ROOT, 'src', 'locales', lang, 'web.json'));
  return { ...common, ...web };
}

function validateKeys(merged) {
  for (const lang of LANGS) {
    const data = merged[lang];
    for (const [key, value] of Object.entries(data)) {
      if (!value || value.trim() === '') {
        console.error(`[i18n] Empty value for key "${key}" in ${lang}`);
        process.exit(1);
      }
      if (value === key) {
        console.error(`[i18n] Untranslated key "${key}" in ${lang} (value equals key)`);
        process.exit(1);
      }
    }
  }
}

function validateSymmetry(merged) {
  const keysByLang = {};
  for (const lang of LANGS) {
    keysByLang[lang] = new Set(Object.keys(merged[lang]));
  }
  const allKeys = new Set([...keysByLang[LANGS[0]], ...keysByLang[LANGS[1]]]);
  for (const key of allKeys) {
    for (const lang of LANGS) {
      if (!keysByLang[lang].has(key)) {
        console.error(`[i18n] Key "${key}" exists in some languages but missing in ${lang}`);
        process.exit(1);
      }
    }
  }
}

function validatePlaceholders(merged) {
  for (const key of Object.keys(merged[LANGS[0]])) {
    const placeholders = new Set();
    for (const lang of LANGS) {
      const value = merged[lang][key];
      const matches = value.match(/\{(\w+)\}/g) || [];
      for (const m of matches) {
        placeholders.add(m);
      }
    }
    for (const lang of LANGS) {
      const value = merged[lang][key];
      for (const ph of placeholders) {
        if (!value.includes(ph)) {
          console.error(
            `[i18n] Placeholder ${ph} mismatch in key "${key}" (${lang}: "${value}")`
          );
          process.exit(1);
        }
      }
    }
  }
}

function generateTypes(merged) {
  const keys = Object.keys(merged[LANGS[0]]).sort();
  const union = keys.map((k) => `  | '${k}'`).join('\n');
  return [
    '// Auto-generated by scripts/merge-locales.mjs — DO NOT EDIT',
    `// Generated at: ${new Date().toISOString()}`,
    '',
    `export type TranslationKey =`,
    union + ';',
    '',
  ].join('\n');
}

function main() {
  mkdirSync(OUT_DIR, { recursive: true });

  const merged = {};
  for (const lang of LANGS) {
    merged[lang] = mergeLocales(lang);
    const outPath = join(OUT_DIR, `${lang}.json`);
    writeFileSync(outPath, JSON.stringify(merged[lang], null, 2) + '\n');
    console.log(`[i18n] Written: ${outPath} (${Object.keys(merged[lang]).length} keys)`);
  }

  // Validate
  validateKeys(merged);
  validateSymmetry(merged);
  validatePlaceholders(merged);

  // Generate types
  const typesContent = generateTypes(merged);
  writeFileSync(join(OUT_DIR, 'keys.ts'), typesContent);
  console.log(`[i18n] Generated types: ${join(OUT_DIR, 'keys.ts')}`);
}

main();
```

```typescript
import { parseAcceptLanguage } from './locale';

const translations: Record<string, TranslationMap> = { zh, en };
const SUPPORTED_LANGS = ['zh', 'en'] as const;
export type SupportedLang = (typeof SUPPORTED_LANGS)[number];

export const FALLBACK_LANG: SupportedLang = 'en';

export function isSupportedLang(lang: string): lang is SupportedLang {
  return SUPPORTED_LANGS.includes(lang as SupportedLang);
}

export function loadTranslations(lang: SupportedLang): TranslationMap {
  return translations[lang] ?? translations[FALLBACK_LANG];
}

export function createTranslateFn(map: TranslationMap): TranslateFn {
  return (key: string, params?: Record<string, string | number>) => {
    let template = map[key];
    if (!template) {
      // Fallback chain: target → en → key itself
      template = translations[FALLBACK_LANG][key];
    }
    if (!template) {
      if (import.meta.env.DEV) {
        console.warn(`[i18n] Missing translation: "${key}"`);
        return `⚠ ${key}`;
      }
      return key;
    }
    if (!params) return template;
    return template.replace(/\{(\w+)\}/g, (_, name: string) =>
      String(params[name] ?? `{${name}}`)
    );
  };
}

export function detectLanguage(
  urlLang: string | undefined,
  cookieLang: string | null,
  acceptLanguage: string | null
): SupportedLang {
  // 1. URL prefix takes highest priority
  if (urlLang && isSupportedLang(urlLang)) return urlLang;

  // 2. Cookie preference
  if (cookieLang && isSupportedLang(cookieLang)) return cookieLang;

  // 3. Browser Accept-Language
  const preferred = parseAcceptLanguage(acceptLanguage);
  if (preferred && isSupportedLang(preferred)) return preferred;

  // 4. Fallback
  return FALLBACK_LANG;
}
```

```typescript
import type { SupportedLang } from './server';

const LANG_COOKIE = 'lang';
const COOKIE_MAX_AGE = 365 * 24 * 60 * 60; // 1 year

export function parseAcceptLanguage(header: string | null): string | null {
  if (!header) return null;
  try {
    // Simple parser: "zh-CN,zh;q=0.9,en;q=0.8" → "zh"
    const first = header.split(',')[0];
    if (!first) return null;
    const tag = first.split(';')[0].trim();
    if (tag.includes('-')) return tag.split('-')[0].toLowerCase();
    return tag.toLowerCase();
  } catch {
    return null;
  }
}

export function setLangCookie(lang: SupportedLang): string {
  return `${LANG_COOKIE}=${lang}; Path=/; Max-Age=${COOKIE_MAX_AGE}; SameSite=Lax`;
}

export function buildSwitchUrl(currentPath: string, targetLang: SupportedLang): string {
  // Replace URL lang prefix: /zh/articles/ → /en/articles/
  const parts = currentPath.replace(/\/+$/, '').split('/').filter(Boolean);
  if (parts.length > 0 && ['zh', 'en'].includes(parts[0])) {
    parts[0] = targetLang;
    return '/' + parts.join('/') + '/';
  }
  return `/${targetLang}/`;
}
```

### Task 19: Create CI check script

**Files:**
- Create: `web/scripts/i18n-check.mjs`

- [ ] **Step 1: Create `web/scripts/i18n-check.mjs`**

```javascript
// Lightweight CI-only check — delegates to merge script's validation logic
import { execSync } from 'node:child_process';
import { existsSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = join(__dirname, '..');

// Check that locale source files exist
const langs = ['zh', 'en'];
const repoRoot = join(ROOT, '..', '..');

let errors = 0;

// Check common locale files
for (const lang of langs) {
  const commonPath = join(repoRoot, 'locales', lang, 'common.json');
  if (!existsSync(commonPath)) {
    console.error(`[i18n-check] Missing: ${commonPath}`);
    errors++;
  }
}

// Check web locale files
for (const lang of langs) {
  const webPath = join(ROOT, 'src', 'locales', lang, 'web.json');
  if (!existsSync(webPath)) {
    console.error(`[i18n-check] Missing: ${webPath}`);
    errors++;
  }
}

// Run merge (which also validates keys, symmetry, placeholders)
try {
  execSync('node scripts/merge-locales.mjs', {
    cwd: ROOT,
    stdio: 'pipe',
  });
} catch (e) {
  console.error('[i18n-check] Merge/validation failed');
  errors++;
}

if (errors > 0) {
  console.error(`[i18n-check] ${errors} error(s) found`);
  process.exit(1);
}

console.log('[i18n-check] All checks passed');
```
